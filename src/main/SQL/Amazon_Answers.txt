1. Highest Salary Per Department
Question

Return employees having highest salary in each department.

Solution
WITH cte AS (
    SELECT *,
           DENSE_RANK() OVER(
               PARTITION BY department
               ORDER BY salary DESC
           ) rnk
    FROM Employee
)
SELECT employee_id,
       department,
       salary
FROM cte
WHERE rnk = 1;

3. Deduplicate Latest Customer Record
Solution
WITH cte AS (
    SELECT *,
           ROW_NUMBER() OVER(
               PARTITION BY customer_id
               ORDER BY updated_ts DESC
           ) rn
    FROM Customer
)
SELECT *
FROM cte
WHERE rn = 1;
4. Running Total
Solution
SELECT
    customer_id,
    txn_date,
    amount,
    SUM(amount) OVER(
        PARTITION BY customer_id
        ORDER BY txn_date
    ) AS running_total
FROM Transactions;
5. Previous Transaction
Solution
SELECT
    customer_id,
    txn_date,
    amount,
    LAG(amount) OVER(
        PARTITION BY customer_id
        ORDER BY txn_date
    ) AS prev_amount
FROM Transactions;
6. Day-1 Retention
Solution
SELECT COUNT(DISTINCT a.user_id) AS retained_users
FROM Activity a
JOIN Activity b
  ON a.user_id = b.user_id
 AND b.activity_date =
     DATE_ADD(a.activity_date, INTERVAL 1 DAY);
7. Customers Without Orders
Solution
SELECT
    c.customer_id,
    c.name
FROM Customers c
LEFT JOIN Orders o
  ON c.customer_id = o.customer_id
WHERE o.customer_id IS NULL;

Alternative:

SELECT *
FROM Customers c
WHERE NOT EXISTS (
    SELECT 1
    FROM Orders o
    WHERE o.customer_id = c.customer_id
);
8. Consecutive Login Days (3 Days)
Solution
WITH cte AS (
    SELECT
        user_id,
        login_date,
        ROW_NUMBER() OVER(
            PARTITION BY user_id
            ORDER BY login_date
        ) rn
    FROM Logins
),
grp AS (
    SELECT *,
           DATE_SUB(login_date, INTERVAL rn DAY) grp_key
    FROM cte
)
SELECT user_id
FROM grp
GROUP BY user_id, grp_key
HAVING COUNT(*) >= 3;
9. Sessionization (30 Min Gap)
Solution
WITH cte AS (
    SELECT *,
           LAG(event_time) OVER(
               PARTITION BY user_id
               ORDER BY event_time
           ) prev_time
    FROM Events
),
flagged AS (
    SELECT *,
           CASE
               WHEN prev_time IS NULL
                    OR TIMESTAMPDIFF(
                         MINUTE,
                         prev_time,
                         event_time
                       ) > 30
               THEN 1
               ELSE 0
           END session_start
    FROM cte
)
SELECT *,
       SUM(session_start) OVER(
           PARTITION BY user_id
           ORDER BY event_time
       ) AS session_id
FROM flagged;
10. Funnel Analysis
Solution
SELECT
    COUNT(DISTINCT CASE
        WHEN event = 'View_Product'
        THEN user_id
    END) AS viewed,

    COUNT(DISTINCT CASE
        WHEN event = 'Add_To_Cart'
        THEN user_id
    END) AS carted,

    COUNT(DISTINCT CASE
        WHEN event = 'Purchase'
        THEN user_id
    END) AS purchased
FROM Events;

12. Monthly Revenue
SELECT
    DATE_FORMAT(order_date,'%Y-%m') AS month,
    SUM(amount) AS revenue
FROM Orders
GROUP BY DATE_FORMAT(order_date,'%Y-%m');
13. Top Customer Per Month
WITH monthly AS (
    SELECT
        DATE_FORMAT(order_date,'%Y-%m') month,
        customer_id,
        SUM(amount) revenue
    FROM Orders
    GROUP BY 1,2
),
ranked AS (
    SELECT *,
           DENSE_RANK() OVER(
               PARTITION BY month
               ORDER BY revenue DESC
           ) rnk
    FROM monthly
)
SELECT *
FROM ranked
WHERE rnk = 1;

15. Immediate Food Delivery II (Amazon-style)
WITH first_orders AS (
    SELECT *,
           ROW_NUMBER() OVER(
               PARTITION BY customer_id
               ORDER BY order_date
           ) rn
    FROM Delivery
)
SELECT ROUND(
       100.0 *
       SUM(
           CASE
             WHEN order_date =
                  customer_pref_delivery_date
             THEN 1
             ELSE 0
           END
       ) / COUNT(*),
       2
) AS immediate_percentage
FROM first_orders
WHERE rn = 1;


Memorize These 5 Patterns
ROW_NUMBER()
DENSE_RANK()
LAG()
SUM() OVER()
LEFT JOIN ... IS NULL

If you can write these from memory, you'll be able to solve most Amazon DE2 SQL questions with minor modifications.
